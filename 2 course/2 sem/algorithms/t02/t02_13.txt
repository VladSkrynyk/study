       | T(n - 1) + O(1),   n >= 1.

T(n) <= T(n - 1) + C <= T(n - 2) + 2 C <=
     <= T(n - 3) + 3 C <= ... <= C n = O(n)

       | T(n - a) + O(1),   n > a.

T(n) <= T(n - a) + C <= T(n - 2a) + 2 C <=
     <= T(n - 3a) + 3 C <= ... <= 2 * (n / 2) C  = O(n)
